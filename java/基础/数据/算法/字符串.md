# 判断字符串是否是回文字符
什么是回文字符串？
"absba"： true
"a": true
"ranko": false

```js
/**
 * 解题思路： 双指针解题
 */
function judge(str){
    let start = 0;
    let end = str.length - 1;
    while(start < end){
        if(str.charAt(start) === str.charAt(end)){
            start++;
            end--; 
        } else {
            return false;
        }
    }

    return true;
}

console.log(judge("abcba"));
```
