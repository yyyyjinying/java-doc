# 常见错误
## invalid character ->  `dep_mp`
INSERT INTO `dep_mp`(user_id,name) values(1000, 'zzz');
## column not allowed here ->  “”
INSERT INTO `dep_mp`(user_id,name) values(1000, “zzz”);
##  missing SELECT keyword  -> 'user_id'
INSERT INTO dep_mp('user_id',name) values(1000, 'zzz');
## column not allowed here -> "bbb"
insert into dep_mp values(1003,"bbb");
